# 在优先队列中自定义比较函数

## 通过 struct 重载()操作符，定义了一个函数对象

~~~c++
struct cmp{
	bool operator()(node a,node b)
	{
		if(a.l != b.l)
            return a.l > b.l;
        return a.r > b.r;
	}
};
priority_queue<node,vector<node>,cmp> heap;
~~~

## 自定义比较函数

~~~~c++
bool cmp(node a, node b) {
    if (a.l != b.l)
        return a.l > b.l;
    return a.r > b.r;
}
priority_queue<node,vector<node>,decltype(&cmp)> heap(cmp); //小顶堆
~~~~

struct Node{

    {
        if(l != w.l)

priority_queue<Node,vector<Node>,less<Node> >heap;
