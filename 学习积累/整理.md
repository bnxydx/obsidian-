# 组合数

~~~
#include <iostream>
#include <algorithm>

using namespace std;
typedef long long LL;

const int N = 100010,mod = 1e9 + 7;
int fact[N];
int infact[N];
LL q_pow(int a,int k,int p)
{
    LL res = 1;
    while(k)
    {
        if(k & 1) res = (LL) res * a % mod;
        a =(LL) a * a % mod;
        k >>= 1;
    }
    return res;
}
int main()
{
	fact[0] = infact[0] = 1;
	for(int i = 1 ;i < N; i ++)
	{
		fact[i] = (LL)fact[i - 1] * i % mod;
		infact[i] = (LL)infact[i - 1] * q_pow(i,mod - 2,mod) % mod;
	}
	int n;
	cin>>n;
	while(n --)
	{
		int a,b;
		cin>>a>>b;
		cout<<(LL)fact[a] * infact[b] % mod * infact[a - b] % mod<<'\n';
	}
	return 0;
}
~~~

在STL中，find函数用于在容器中查找指定值的元素，而find_if函数用于根据指定条件查找元素。在您的代码中，您需要根据第一个元素（pair中的first）的值来查找元素，而不是直接比较元素是否相等，因此需要使用find_if函数。

find函数只能用于***查找与指定值相等的元素***，而find_if函数可以根据***自定义***的条件来查找元素。在这种情况下，您需要使用find_if函数并提供一个lambda表达式作为条件，以便根据第一个元素的值来查找元素。

# for循环控制奇数列遍历

~~~
给出1~n的一个排列，统计该排列有多少个长度为奇数的连续子序列的中位数是b。中位数是指把所有元素从小到大排列后，位于中间的数。
输入描述:
	第一行为两个正整数n和b ，第二行为1~n 的排列。
输出描述:
	输出一个整数，即中位数为b的连续子序列个数。
示例1
输入
7 4
5 7 2 4 3 1 6
输出
4

#include<iostream>

using namespace std;

const int N = 1010000;
int n,m;
int sum[N];
int a[N];
int main(){
    cin>>n>>m;
    
    int k=0;
    for(int i=1;i<=n;i++)
    {
        int x;
        cin>>x;
        if(x>m) sum[i] = 1;
        else if (x<m) sum[i] = -1;
        else k = i; 
        sum[i] += sum[i-1];
    }
    int res = 0;
    for(int i=1;i<=k;i++)
        for(int j = (k - i) % 2==0 ?k:k + 1;j<=n;j+=2)
        {
            if(sum[j] - sum[i-1]==0) res++;
        }
    cout<<res;
    return 0;
    
}
