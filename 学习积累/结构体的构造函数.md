

~~~c++
struct node{
	int p1,p2;
	node(int n1,int n2)
	{
		p1 = n1;
		p2 = n2;
	}
}
int main(){
	node a(2,3);
	/*
	a.p1 = 2;
	a.p2 = 3;
	*/
}
~~~

~~~c++
struct node{
	int p1,p2;
	node(int n1,int n2) :
    	p1(n1),p2(n2){}
};//构造了int类型的n1，n2，并将值赋值给了结构体里面的对应成员
~~~



- 如果写成如下，计算机无法识别

- ~~~
	node a;
	a = node(2,3);
	~~~

- 则在结构体中应该加一句构造函数的声明

- ~~~c++
	struct node{
		int p1,int p2;
		node(){}
		node(int n1,int n2){
			p1 = n1,p2 = n2;
		}
	};
	node a = node(2,3);
	~~~



- 建议写构造函数前都要写

- ~~~c++
	node(){}
	~~~

- ~~~c++
	struct node {
		int n1, n2;
		int a[10];
		node() {
			p1 = 2;
			memset( a, 0, sizeof( a ) );
		}
		node( int n1, int n2 ) {
			p1 = n1, p2 = n2;
		}
	}a;
	
	~~~

- 这段里面的内容意思就是一旦你申请了一个结构体a，计算机自动的就完成了a . p 1 = 2 a的赋值以及结构体嵌套数组的清零操作，不用自己在里面手打一个clear函数就能起到同样的效果
	经常在矩阵快速幂等各种场合的时候使用
