# 问卷

- 目前单选和多选

~~~
from selenium.webdriver.common.by import By
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
import random
import time
from fake_useragent import UserAgent

headers = {"User-Agent": UserAgent().chrome}
options = Options()
options.add_argument(f'user-agent={headers}')
driver = webdriver.Chrome()

def q_num():
    """
    获取题目数量

    :return: int
    """
    fieldset = driver.find_element(By.ID, 'fieldset1')
    questions = fieldset.find_elements(By.CSS_SELECTOR, 'div.field.ui-field-contain')
    question_count = len(questions)
    print(f"题目数量: {question_count}")
    return question_count

def q_select(num):
    """

    :param num: 题目数量
    :return:
    """
    for i in range(1,num + 1):
        q = driver.find_element(By.ID,f'div{i}')
        # 单选题
        options = q.find_elements(By.CLASS_NAME, 'ui-radio')
        # 多选题
        options2 = q.find_elements(By.CLASS_NAME, 'ui-checkbox')
        if options:
            random_option = random.choice(options)
            label = random_option.find_element(By.CLASS_NAME, 'label')
            label.click()
            time.sleep(0.5)
            print(f"问题 {i}: 已随机选择选项")
            continue
        if options2:
            random_option = random.choice(options2)
            label = random_option.find_element(By.CLASS_NAME, 'label')
            label.click()
            time.sleep(0.5)
            print(f"问题 {i}: 已随机选择选项")
            continue
        else:
            print(f"问题 {i}: 未找到选项")

def zonghe(times):
    num = q_num()
    for i in range(0, times):
        # 问卷网址
        driver.get('https://www.wjx.cn/vm/mk2UxmY.aspx')
        q_select(num)
        time.sleep(random.randint(0, 1))
        driver.find_element(By.CSS_SELECTOR,'#ctlNext').click()
        time.sleep(4)
        print('已经提交了{}次问卷'.format(int(i) + int(1)))
        time.sleep(4)
        driver.quit()
        time.sleep(1)

if __name__ == "__main__":
    # 填几次
    zonghe(1)

~~~
