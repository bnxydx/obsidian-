class LeNet5(nn.Module):
    """
    经典 LeNet-5 网络（适配任意单通道/三通道输入）

    网络层说明：
        C1  - 卷积层1 : in_channels → 6,  kernel=5, stride=1, padding=2  → 特征图 6×H×W
        S2  - 池化层1 : 平均池化,   kernel=2, stride=2                   → 特征图 6×(H/2)×(W/2)
        C3  - 卷积层2 : 6 → 16,    kernel=5, stride=1                   → 特征图 16×(H/2-4)×(W/2-4)
        S4  - 池化层2 : 平均池化,   kernel=2, stride=2                   → 特征图 16×...×...
        C5  - 全卷积层: 16 → 120,  kernel=5（展平为向量）
        F6  - 全连接层: 120 → 84
        OUT - 输出层  : 84 → num_classes
    """
    def __init__(self, in_channels: int = 1, num_classes: int = 10):
        super(LeNet5, self).__init__()

        # ── 特征提取部分 ──────────────────────────────────────────
        self.features = nn.Sequential(
            # C1: 卷积 + 激活  (padding=2 保持 32x32 → 32x32)
            nn.Conv2d(in_channels, 6, kernel_size=5, stride=1, padding=2),
            nn.Tanh(),
            # S2: 平均池化  32x32 → 16x16
            nn.AvgPool2d(kernel_size=2, stride=2),

            # C3: 卷积 + 激活  16x16 → 12x12
            nn.Conv2d(6, 16, kernel_size=5, stride=1),
            nn.Tanh(),
            # S4: 平均池化  12x12 → 6x6
            nn.AvgPool2d(kernel_size=2, stride=2),

            # C5: 卷积（相当于全连接）  6x6 → 2x2 for CIFAR32；对MNIST 28→14→10→5→1
            nn.Conv2d(16, 120, kernel_size=5, stride=1),
            nn.Tanh(),
        )

        # 自动推断 flatten 后的维度（不需要手动计算）
        self._flatten_dim = self._get_flatten_dim(in_channels)

        # ── 分类器部分 ────────────────────────────────────────────
        self.classifier = nn.Sequential(
            # F6
            nn.Linear(self._flatten_dim, 84),
            nn.Tanh(),
            # OUTPUT
            nn.Linear(84, num_classes),
        )

    def _get_flatten_dim(self, in_channels: int) -> int:
        """用一个假输入前向传播，自动得到 flatten 维度"""
        with torch.no_grad():
            # CIFAR-10 输入 32×32，MNIST 输入 28×28；统一用 32×32 做探测
            dummy = torch.zeros(1, in_channels, 32, 32)
            out   = self.features(dummy)
            return int(out.view(1, -1).shape[1])

    def forward(self, x: torch.Tensor):
        x = self.features(x)           # 卷积特征提取
        x = x.view(x.size(0), -1)      # 展平
        x = self.classifier(x)         # 全连接分类
        return x

def train_one_epoch(model, loader, criterion, optimizer, device):
    model.train()
    total_loss, correct, total = 0.0, 0, 0
    for images, labels in loader:
        images, labels = images.to(device), labels.to(device)
        optimizer.zero_grad()
        outputs = model(images)
        loss    = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        total_loss += loss.item() * labels.size(0)
        _, predicted = outputs.max(1)
        correct += predicted.eq(labels).sum().item()
        total   += labels.size(0)

    return total_loss / total, 100.0 * correct / total


@torch.no_grad()
def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct, total = 0.0, 0, 0
    for images, labels in loader:
        images, labels = images.to(device), labels.to(device)
        outputs = model(images)
        loss    = criterion(outputs, labels)

        total_loss += loss.item() * labels.size(0)
        _, predicted = outputs.max(1)
        correct += predicted.eq(labels).sum().item()
        total   += labels.size(0)

    return total_loss / total, 100.0 * correct / total

def main():
    # 1. 加载数据
    train_loader, test_loader, in_channels, num_classes = \
        get_dataloaders(TASK, BATCH_SIZE)

    # 2. 构建模型
    model = LeNet5(in_channels=in_channels, num_classes=num_classes).to(DEVICE)
    print(f"\n{'='*55}")
    print(f"  LeNet-5 网络结构  (任务: {TASK.upper()})")
    print(f"{'='*55}")
    print(model)

    # 统计参数量
    total_params = sum(p.numel() for p in model.parameters())
    train_params = sum(p.numel() for p in model.parameters() if p.requires_grad)
    print(f"\n  总参数量:    {total_params:,}")
    print(f"  可训练参数:  {train_params:,}")
    print(f"{'='*55}\n")

    # 3. 损失函数 & 优化器 & 学习率调度
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=LR, weight_decay=1e-4)
    scheduler = optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=NUM_EPOCHS)

    # 4. 训练循环
    history = {'train_loss': [], 'train_acc': [], 'test_loss': [], 'test_acc': []}
    best_acc = 0.0
    best_model_path = os.path.join(SAVE_DIR, f'lenet5_{TASK}_best.pth')

    print(f"开始训练，共 {NUM_EPOCHS} 轮 ...")
    for epoch in range(1, NUM_EPOCHS + 1):
        t0 = time.time()

        train_loss, train_acc = train_one_epoch(model, train_loader, criterion, optimizer, DEVICE)
        test_loss,  test_acc  = evaluate(model, test_loader, criterion, DEVICE)
        scheduler.step()

        history['train_loss'].append(train_loss)
        history['train_acc'].append(train_acc)
        history['test_loss'].append(test_loss)
        history['test_acc'].append(test_acc)

        elapsed = time.time() - t0
        print(f"Epoch [{epoch:02d}/{NUM_EPOCHS}]  "
              f"Train Loss: {train_loss:.4f}  Train Acc: {train_acc:.2f}%  |  "
              f"Test Loss: {test_loss:.4f}  Test Acc: {test_acc:.2f}%  "
              f"({elapsed:.1f}s)")

        # 保存最优模型
        if test_acc > best_acc:
            best_acc = test_acc
            torch.save(model.state_dict(), best_model_path)

    print(f"\n✅ 训练完成！最佳测试准确率: {best_acc:.2f}%")
    print(f"   最优模型已保存至: {best_model_path}")

    # 5. 结果保存
    result_txt = os.path.join(SAVE_DIR, f'lenet5_{TASK}_result.txt')
    with open(result_txt, 'w', encoding='utf-8') as f:
        f.write(f"任务:           {TASK.upper()}\n")
        f.write(f"批次大小:       {BATCH_SIZE}\n")
        f.write(f"学习率:         {LR}\n")
        f.write(f"训练轮数:       {NUM_EPOCHS}\n")
        f.write(f"最佳测试准确率: {best_acc:.2f}%\n")
        f.write(f"最终测试Loss:   {history['test_loss'][-1]:.6f}\n")
    print(f"   实验结果已保存至: {result_txt}")

    # 6. 绘制训练曲线
    plot_curves(history, TASK, SAVE_DIR)


if __name__ == '__main__':
    main()

class EarlyStopping:
    """
    监控测试集准确率，当连续 patience 轮不再提升时触发停止。
    同时保存最优模型权重，停止后自动恢复。
    """
    def __init__(self, patience: int = 10, model_path: str = 'best.pth'):
        self.patience   = patience
        self.model_path = model_path
        self.counter    = 0          # 连续未提升轮数
        self.best_acc   = 0.0        # 历史最佳准确率
        self.stop       = False      # 是否触发停止

    def step(self, acc: float, model: nn.Module) -> bool:
        """每轮结束后调用，返回 True 表示应停止训练。"""
        if acc > self.best_acc:
            self.best_acc = acc
            self.counter  = 0
            torch.save(model.state_dict(), self.model_path)  # 保存最优权重
        else:
            self.counter += 1
            if self.counter >= self.patience:
                self.stop = True
        return self.stop

    def load_best(self, model: nn.Module):
        """训练结束后恢复最优权重。"""
        model.load_state_dict(torch.load(self.model_path, map_location='cpu'))
        print(f"[EarlyStopping] 已恢复最优模型，最佳测试准确率: {self.best_acc:.2f}%")

class FaceResNet18(nn.Module):
    """
    适配 Olivetti Faces 的 ResNet18
    ─────────────────────────────────────────────────────
    输入:  (B, 1, 64, 64)
    输出:  (B, 40)
    """

    def __init__(self, num_classes: int = 40):
        super(FaceResNet18, self).__init__()

        # 加载预训练模型或初始化
        # weights=None 表示随机初始化，weights=models.ResNet18_Weights.DEFAULT 表示使用 ImageNet 预训练权重
        self.resnet = models.resnet18(weights=None)

        # 修改第一层卷积：由 3 通道改为 1 通道
        # ResNet18 的第一层是 conv1: Conv2d(3, 64, kernel_size=7, stride=2, padding=3, bias=False)
        self.resnet.conv1 = nn.Conv2d(1, 64, kernel_size=7, stride=2, padding=3, bias=False)

        # 修改全连接层：由 1000 类改为 40 类
        # ResNet18 的最后一层是 fc: Linear(in_features=512, out_features=1000, bias=True)
        num_ftrs = self.resnet.fc.in_features
        self.resnet.fc = nn.Linear(num_ftrs, num_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.resnet(x)

def get_dataloaders(batch_size: int):
    """
    使用 fetch_olivetti_faces 读取本地 olivetti_py3.pkz
    返回 train_loader, test_loader
    数据集按 8:2 划分训练集与测试集
    """
    print("[INFO] 正在加载 Olivetti Faces 数据集…")
    faces = fetch_olivetti_faces(data_home=DATA_PATH)
    # images: (400, 64, 64) float32, 值域 [0,1]
    # target: (400,) int，0~39
    images  = faces.images                               # (400, 64, 64)
    targets = faces.target                               # (400,)

    # 转为 Tensor，增加通道维度 → (400, 1, 64, 64)
    X = torch.tensor(images, dtype=torch.float32).unsqueeze(1)
    y = torch.tensor(targets, dtype=torch.long)

    full_dataset = TensorDataset(X, y)

    # 8:2 划分
    total      = len(full_dataset)                       # 400
    train_size = int(total * 0.8)                        # 320
    test_size  = total - train_size                      # 80

    train_ds, test_ds = random_split(
        full_dataset, [train_size, test_size],
        generator=torch.Generator().manual_seed(42)
    )

    train_loader = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
    test_loader  = DataLoader(test_ds,  batch_size=batch_size, shuffle=False)

    print(f"[INFO] 图像尺寸: {images.shape[1]}×{images.shape[2]}  类别数: {len(np.unique(targets))}")
    print(f"[INFO] 训练集: {train_size} 张  测试集: {test_size} 张")
    return train_loader, test_loader

        'test_loss':  [],
        'test_acc':   [],
        'lr':         []
    }
    best_model_path = os.path.join(SAVE_DIR, 'facecnn_olivetti_best_res.pth')
    early_stop      = EarlyStopping(patience=PATIENCE, model_path=best_model_path)

    print(f"开始训练，共 {NUM_EPOCHS} 轮  设备: {DEVICE}  早停耐心: {PATIENCE} 轮 ...")
    for epoch in range(1, NUM_EPOCHS + 1):
        t0 = time.time()

        current_lr = optimizer.param_groups[0]['lr']

        train_loss, train_acc = train_one_epoch(model, train_loader, criterion, optimizer, DEVICE)
        test_loss,  test_acc  = evaluate(model, test_loader, criterion, DEVICE)
        scheduler.step()

        history['train_loss'].append(train_loss)
        history['train_acc'].append(train_acc)
        history['test_loss'].append(test_loss)
        history['test_acc'].append(test_acc)
        history['lr'].append(current_lr)

        elapsed = time.time() - t0
        improved = '★' if test_acc > early_stop.best_acc else f'(patience {early_stop.counter+1}/{PATIENCE})'
        print(f"Epoch [{epoch:02d}/{NUM_EPOCHS}]  LR: {current_lr:.6f}  "
              f"Train Loss: {train_loss:.4f}  Train Acc: {train_acc:.2f}%  |  "
              f"Test Loss: {test_loss:.4f}  Test Acc: {test_acc:.2f}%  "
              f"{improved}  ({elapsed:.1f}s)")

        # 早停检查（内部自动保存最优模型）
        if early_stop.step(test_acc, model):
            print(f"\n⏹  早停触发！测试准确率已连续 {PATIENCE} 轮未提升，停止训练。")
            break

    # 恢复最优权重
    early_stop.load_best(model)
    print(f"\n✅ 训练完成！最佳测试准确率: {early_stop.best_acc:.2f}%")
    print(f"   最优模型已保存至: {best_model_path}")

    # 5. 结果保存
    result_txt = os.path.join(SAVE_DIR, 'facecnn_olivetti_result_res.txt')
    with open(result_txt, 'w', encoding='utf-8') as f:
        f.write(f"任务:           Olivetti Faces 人脸识别\n")
        f.write(f"批次大小:       {BATCH_SIZE}\n")
        f.write(f"学习率:         {LR}\n")
        f.write(f"训练轮数:       {NUM_EPOCHS}\n")
        f.write(f"最佳测试准确率: {early_stop.best_acc:.2f}%\n")
        f.write(f"最终测试Loss:   {history['test_loss'][-1]:.6f}\n")
    print(f"   实验结果已保存至: {result_txt}")

    # 6. 绘制训练曲线
    plot_curves(history, SAVE_DIR)


if __name__ == '__main__':
    main()

~~~

**早停机制（Early Stopping）：** * GAN 不能看 Loss 来做早停，因为生成器和判别器的 Loss 在理想状态下是处于动态博弈、上下波动的，Loss 变低不等于画得好。

-   **这里的做法：** 引入了 `FID (Fréchet Inception Distance)` 的平替思想——**监控判别器对真图的打分稳定性**。如果判别器连续 20 个 Epoch 彻底压制生成器（或者生成器完全崩坏，导致两者 Loss 极度失衡），代码会自动触发 `Early Stopping`，帮你节省算力和时间。

**优化器（Optimizer）：** * 沿用了 StyleGAN 论文中最经典的 **Adam 优化器**，并专门将 $\beta_1$ 设为 `0.0`，$\beta_2$ 设为 `0.99`。这是 GAN 训练的行业标准，能有效防止对抗训练时产生剧烈的震荡。

**余弦退火学习率（Cosine Annealing LR）：**

-   引入了 `CosineAnnealingLR` 动态调整策略。学习率会从初始的 `2e-4` 随着余弦曲线平滑下降，在训练后期通过微弱的学习率让模型精细雕刻蕾姆的眼睛、睫毛等面部细节。

**数据增强（Augmentation）：**

-   代码中原本就自带了**全套的自适应数据增强（ADA）**。它会根据过拟合程度，自动、动态地对蕾姆的头像进行水平翻转、随机轻微缩放、以及亮度和对比度的微调。

**Loss 曲线看板绘制：**

-   训练结束后，代码会在根目录下自动生成一张 `gan_loss_curve.png` 趋势图，帮你复盘生成器（G）与判别器（D）的博弈过程。





### 2. 方案二：如果你使用高级架构 + 现代增强技术（如 StyleGAN2-ADA）

数据集作者创建这个数据集的初衷，其实是为了配合 NVIDIA 的 **StyleGAN2-ADA（Adaptive Discriminator Augmentation，自适应判别器数据增强）** 算法。这是目前工业界针对“小数据集”的最佳解决方案。

如果你用 StyleGAN2-ADA，并开启数据增强，从头训练：

-   **能达到的程度：**
    -   **中等清晰度：** 能生成分辨率在 $256 \times 256$ 或 $512 \times 512$ 左右、相对清晰的蕾姆头像。
    -   **表情与画风多样：** 因为数据来源于不同画师的 Pixiv 投稿，你能看到不同画风（Q版、写实动漫风、水彩风）的蕾姆，甚至能学到不同的表情（哭泣、微笑、黑化）。
    -   **偶尔有伪影：** 偶尔会出现多出一只眼睛、头发线条断裂或背景扭曲的小瑕疵，但整体已经具备很高的观赏性。
